15 def __init__(self): 

16 """sqlite connection initialization.""" 

17 try: 

18 with open("aims/config.yaml", 'r') as yaml_file: 

19 cfg = yaml.safe_load(yaml_file) 

20 db_path = cfg['mysql']['db'] 

21 conn = sqlite3.connect(db_path) 

22 except sqlite3.Error as e: 

23 print(e) 

24 sys.exit() 

25 else:

42 def db_script(self): 

43 """this method is used to create schema for aims project. """ 

44 try: 

45 with open('data/schema_script.sql', 'r') as sql_file: 

46 sql_script = sql_file.read() 

47 self.cur.executescript(sql_script) 

48 self.commit_data() 

49 

50 except sqlite3.Error as error: 

51 print('unable to run database script') 

52 print(error) 

53 self.rollback_data() 

54 sys.exit() 

55 finally:
